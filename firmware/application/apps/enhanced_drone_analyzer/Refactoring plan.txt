### Step 3: The Devil's Advocate (Self-Correction)

#### 3.1 Memory Attack

**QUESTION: Will LUTs be placed in Flash or RAM?**

```cpp
// CURRENT CODE:
static constexpr uint32_t COLORS[6] = { ... };  // May end up in RAM!
```

**PROBLEM**: Without explicit section attributes, constexpr arrays may be copied to RAM at startup.

**ATTACK RESOLUTION:**
```cpp
// CORRECT:
#ifdef __GNUC__
    #define FLASH_CONST __attribute__((section(".rodata"))) const
#else
    #define FLASH_CONST const
#endif

static FLASH_CONST uint32_t COLORS[6] = { ... };
```

**VERIFICATION**: Check `.map` file for section placement:
```
.rodata.EDA::Constants::COLORS
    0x08001234    0x18 bytes
```

#### 3.2 Performance Attack

**QUESTION: Is the abstraction free?**

```cpp
// Template-based settings serialization:
template<typename T>
constexpr void serialize_setting(char* buf, size_t& offset, 
                                  const char* key, const T& value);
```

**ATTACK**: Each template instantiation generates separate code for each type T.

**ANALYSIS**: For 6 types (bool, uint32_t, int32_t, uint64_t, string, bitfield):
- Code size per instantiation: ~50 bytes
- Total: 6 × 50 = 300 bytes

**vs CURRENT**: Single function with switch-case: ~200 bytes

**RESOLUTION**: Use `if constexpr` to share common paths:

```cpp
template<typename T>
constexpr void serialize_setting(char* buf, size_t& offset,
                                  const char* key, const T& value) {
    offset += snprintf(buf + offset, MAX_OFFSET - offset, "%s=", key);
    if constexpr (std::is_same_v<T, bool>) {
        offset += snprintf(buf + offset, MAX_OFFSET - offset, "%s", 
                          value ? "true" : "false");
    } else if constexpr (std::is_integral_v<T>) {
        offset += snprintf(buf + offset, MAX_OFFSET - offset, "%lld", 
                          (long long)value);
    }
    buf[offset++] = '\n';
}
```

This reduces generated code to ~100 bytes total.

#### 3.3 Safety Attack

**QUESTION: Am I hiding type conversions with auto?**

```cpp
// CURRENT CODE:
auto read_res = file.read(read_buffer, SettingsLoadBuffer::READ_BUFFER_SIZE);
```

**PROBLEM**: `auto` hides the return type, making error handling unclear.

**RESOLUTION**: Be explicit for error-prone operations:

```cpp
File::ReadResult read_res = file.read(read_buffer, BUFFER_SIZE);
if (read_res.is_error() || read_res.value() == 0) {
    break;
}
```

**QUESTION: Is there potential for integer overflow?**

```cpp
// CURRENT CODE:
uint32_t bin_width_hz = (params.slice_bandwidth_hz * INV_BIN_COUNT_Q16) >> 16;
```

**ANALYSIS**: `slice_bandwidth_hz` is uint32_t (max 28MHz). Multiplication by Q16 fixed-point:
- Max: 28,000,000 × 65,536 = 1,835,008,000,000
- Exceeds uint32_t range!

**RESOLUTION**: Use uint64_t for intermediate calculation:

```cpp
uint32_t bin_width_hz = static_cast<uint32_t>(
    (static_cast<uint64_t>(params.slice_bandwidth_hz) * INV_BIN_COUNT_Q16) >> 16
);
```

#### 3.4 Meyers Attack

**QUESTION: Am I using exceptions? RTTI?**

**SCAN RESULTS:**
- No `try`, `catch`, `throw` found ✓
- No `dynamic_cast`, `typeid` found ✓

**QUESTION: Is this copy-constructible? Should it be?**

```cpp
// CURRENT CODE:
class DetectionRingBuffer {
    DetectionRingBuffer(const DetectionRingBuffer&) = delete;  // ✓
    DetectionRingBuffer& operator=(const DetectionRingBuffer&) = delete;  // ✓

```

```cpp

```

**PROBLEM**: Static mutable data in a class suggests singleton pattern needed, or explicit deletion of copy operations.

**RESOLUTION**: Either make it a true singleton, or make all methods non-static with instance data.

// ============================================================
// STRING UTILITIES (Zero-Heap)
// ============================================================

inline void safe_strcpy(char* dest, const char* src, size_t max_len) noexcept {
    if (!dest || !src || max_len == 0) return;
    size_t i = 0;
    while (i < max_len - 1 && src[i] != '\0') {
        dest[i] = src[i];
        ++i;
    }
    dest[i] = '\0';
}

inline size_t safe_strlen(const char* str, size_t max_len) noexcept {
    if (!str) return 0;
    size_t len = 0;
    while (len < max_len && str[len] != '\0') { ++len; }
    return len;
}

// ============================================================
// FREQUENCY FORMATTING (Zero-Allocation)
// ============================================================

struct FrequencyFormat {
    enum class Style : uint8_t {
        COMPACT_GHZ,    // "2.4G"
        STANDARD_GHZ,   // "2.45GHz"
        STANDARD_MHZ    // "2450MHz"
    };
    
    static void format(char* buf, size_t buf_size, Frequency hz, Style style) noexcept {
        if (!buf || buf_size == 0) return;
        
        switch (style) {
            case Style::COMPACT_GHZ: {
                uint32_t ghz = static_cast<uint32_t>(hz / 1'000'000'000ULL);
                uint32_t dec = static_cast<uint32_t>((hz % 1'000'000'000ULL) / 100'000'000ULL);
                if (dec > 0) {
                    snprintf(buf, buf_size, "%u.%uG", ghz, dec);
                } else {
                    snprintf(buf, buf_size, "%uG", ghz);
                }
                break;
            }
            case Style::STANDARD_GHZ: {
                uint32_t ghz = static_cast<uint32_t>(hz / 1'000'000'000ULL);
                uint32_t dec = static_cast<uint32_t>((hz % 1'000'000'000ULL) / 10'000'000ULL);
                snprintf(buf, buf_size, "%u.%03uGHz", ghz, dec);
                break;
            }
            case Style::STANDARD_MHZ: {
                uint32_t mhz = static_cast<uint32_t>(hz / 1'000'000ULL);
                snprintf(buf, buf_size, "%uMHz", mhz);
                break;
            }
        }
    }
};

// ============================================================
// COLOR CONVERSION (RGB888 to RGB565)
// ============================================================

struct ColorConvert {
    static constexpr uint16_t rgb888_to_rgb565(uint32_t rgb888) noexcept {
        uint8_t r = (rgb888 >> 16) & 0xFF;
        uint8_t g = (rgb888 >> 8) & 0xFF;
        uint8_t b = rgb888 & 0xFF;
        
        // RGB565: rrrrrGGGGGGbbbbb
        return static_cast<uint16_t>(
            ((r & 0xF8) << 8) |  // 5 bits R
            ((g & 0xFC) << 3) |  // 6 bits G
            ((b & 0xF8) >> 3)    // 5 bits B
        );
    }
};

// ============================================================
// MEDIAN FILTER (Template)
// ============================================================

template<typename T, size_t N>
class MedianFilter {
public:
    void add(T value) noexcept {
        window_[head_] = value;
        head_ = (head_ + 1) % N;
        if (head_ == 0) full_ = true;
    }
    
    T get_median() const noexcept {
        if (!full_) return window_[0];
        
        T temp[N];
        for (size_t i = 0; i < N; ++i) temp[i] = window_[i];
        
        // Selection sort up to middle
        size_t k = N / 2;
        for (size_t i = 0; i <= k; ++i) {
            size_t min_idx = i;
            for (size_t j = i + 1; j < N; ++j) {
                if (temp[j] < temp[min_idx]) min_idx = j;
            }
            if (min_idx != i) {
                T tmp = temp[i];
                temp[i] = temp[min_idx];
                temp[min_idx] = tmp;
            }
        }
        return temp[k];
    }
    
    void reset() noexcept {
        full_ = false;
        head_ = 0;
        for (size_t i = 0; i < N; ++i) window_[i] = T{};
    }

private:
    T window_[N] = {};
    size_t head_ = 0;
    bool full_ = false;
};

// ============================================================
// CACHED VALUE HELPER
// ============================================================

template<typename T>
class CachedValue {
public:
    explicit CachedValue(T initial = T{}) : value_(initial), dirty_(true) {}
    
    bool update(T new_value) noexcept {
        if (value_ != new_value) {
            value_ = new_value;
            dirty_ = true;
            return true;
        }
        return false;
    }
    
    const T& get() const noexcept { return value_; }
    bool is_dirty() const noexcept { return dirty_; }
    void clear_dirty() noexcept { dirty_ = false; }

private:
    T value_;
    bool dirty_;
};

} // namespace EDA

#endif // EDA_UTILS_HPP_

// File: eda_spectral.hpp
#ifndef EDA_SPECTRAL_HPP_
#define EDA_SPECTRAL_HPP_

#include "eda_constants.hpp"
#include "eda_utils.hpp"
#include <array>
#include <cstdint>

namespace EDA {

// ============================================================
// SPECTRAL ANALYZER (RAII-Managed State)
// ============================================================

class SpectralAnalyzer {
public:
    struct Result {
        SignalSignature signature = SignalSignature::NOISE;
        uint8_t max_val = 0;
        uint8_t avg_val = 0;
        uint8_t noise_floor = 0;
        uint8_t snr = 0;
        uint32_t signal_width_hz = 0;
        uint8_t width_bins = 0;
        bool is_valid = false;
    };
    
    SpectralAnalyzer() = default;
    
    // Non-copyable (owns histogram state)
    SpectralAnalyzer(const SpectralAnalyzer&) = delete;
    SpectralAnalyzer& operator=(const SpectralAnalyzer&) = delete;
    
    Result analyze(const std::array<uint8_t, 256>& db_buffer,
                   uint32_t slice_bandwidth_hz) noexcept {
        Result result;
        
        // Use local histogram (stack-based, no static state)
        std::array<uint16_t, 128> histogram = {};
        
        uint32_t sum = 0;
        
        // Build histogram
        for (size_t i = SpectralConfig::VALID_BIN_START;
             i < SpectralConfig::VALID_BIN_END; ++i) {
            uint8_t value = db_buffer[i];
            histogram[value / 2]++;
            sum += value;
        }
        
        // Find median (noise floor)
        constexpr size_t MEDIAN_TARGET = SpectralConfig::VALID_BIN_COUNT / 2;
        size_t cumulative = 0;
        
        for (size_t bin = 0; bin < 128; ++bin) {
            cumulative += histogram[bin];
            if (cumulative > MEDIAN_TARGET) {
                result.noise_floor = static_cast<uint8_t>(bin * 2);
                break;
            }
        }
        
        // Calculate average
        result.avg_val = static_cast<uint8_t>(
            sum / SpectralConfig::VALID_BIN_COUNT
        );
        
        // Find max
        for (size_t i = SpectralConfig::VALID_BIN_START;
             i < SpectralConfig::VALID_BIN_END; ++i) {
            if (db_buffer[i] > result.max_val) {
                result.max_val = db_buffer[i];
            }
        }
        
        // Calculate SNR
        result.snr = (result.max_val > result.noise_floor)
            ? (result.max_val - result.noise_floor) : 0;
        
        // Early exit if noise
        if (result.snr < SpectralConfig::SNR_THRESHOLD) {
            result.signature = SignalSignature::NOISE;
            result.is_valid = true;
            return result;
        }
        
        // Count signal width
        uint8_t threshold = (result.max_val > SpectralConfig::PEAK_THRESHOLD_DB)
            ? (result.max_val - SpectralConfig::PEAK_THRESHOLD_DB) : 0;
        
        for (size_t i = SpectralConfig::VALID_BIN_START;
             i < SpectralConfig::VALID_BIN_END; ++i) {
            if (db_buffer[i] >= threshold) {
                result.width_bins++;
            }
        }
        
        // Calculate width in Hz (avoiding overflow)
        uint64_t width_calc = static_cast<uint64_t>(result.width_bins) *
                             static_cast<uint64_t>(SpectralConfig::INV_BIN_COUNT_Q16);
        result.signal_width_hz = static_cast<uint32_t>(
            (width_calc * slice_bandwidth_hz) >> 16
        );
        
        // Classify signal
        result.signature = classify_signal(result.signal_width_hz);
        result.is_valid = true;
        return result;
    }

private:
    static constexpr SignalSignature classify_signal(uint32_t width_hz) noexcept {
        if (width_hz >= SpectralConfig::WIFI_MIN_WIDTH_HZ) {
            return SignalSignature::WIDEBAND_WIFI;
        }
        if (width_hz <= SpectralConfig::DRONE_MAX_WIDTH_HZ) {
            return SignalSignature::NARROWBAND_DRONE;
        }
        return SignalSignature::NARROWBAND_DRONE;
    }
};

} // namespace EDA

#endif // EDA_SPECTRAL_HPP_
